```go
package services

import (
	"errors"
	"time"

	"github.com/google/uuid"
	"gorm.io/gorm"

	"github.com/tachigo/tachigo/internal/models"
)

var (
	ErrRaffleNotFound     = errors.New("raffle not found")
	ErrRaffleWrongStatus  = errors.New("raffle status does not allow this operation")
	ErrNoEligibleEntries  = errors.New("no eligible entries to draw from")
	ErrDrawAlreadyClaimed = errors.New("draw already claimed")
	ErrDrawExpired        = errors.New("draw claim has expired")
	ErrClaimTokenInvalid  = errors.New("invalid or expired claim token")
)

type RaffleService struct {
	db *gorm.DB
}

func NewRaffleService(db *gorm.DB) *RaffleService {
	return &RaffleService{db: db}
}

func (s *RaffleService) Create(
	streamerID uuid.UUID,
	title string,
	source models.RaffleSource,
	filterConfig *string,
	scheduledAt *time.Time,
	claimDeadlineHours int,
) (*models.Raffle, error) {
	raffle := &models.Raffle{
		StreamerID:         streamerID,
		Title:              title,
		Status:             models.RaffleStatusDraft,
		Source:             source,
		FilterConfig:       filterConfig,
		ScheduledAt:        scheduledAt,
		ClaimDeadlineHours: claimDeadlineHours,
	}
	if err := s.db.Create(raffle).Error; err != nil {
		return nil, err
	}
	return raffle, nil
}

func (s *RaffleService) GetByID(id uuid.UUID) (*models.Raffle, error) {
	var raffle models.Raffle
	if err := s.db.Where("id = ?", id).First(&raffle).Error; err != nil {
		if errors.Is(err, gorm.ErrRecordNotFound) {
			return nil, ErrRaffleNotFound
		}
		return nil, err
	}
	return &raffle, nil
}

func (s *RaffleService) ListByStreamer(streamerID uuid.UUID) ([]models.Raffle, error) {
	var raffles []models.Raffle
	if err := s.db.
		Where("streamer_id = ?", streamerID).
		Order("created_at DESC").
		Find(&raffles).Error; err != nil {
		return nil, err
	}
	return raffles, nil
}
```

```go
// ImportCSV reads a CSV with columns: twitch_user_id, display_name, tier.
// Only rows whose twitch_user_id matches an existing auth_provider are imported.
// Returns (imported count, skipped count, error).
func (s *RaffleService) ImportCSV(raffleID uuid.UUID, r io.Reader) (int, int, error) {
	raffle, err := s.GetByID(raffleID)
	if err != nil {
		return 0, 0, err
	}
	if raffle.Status != models.RaffleStatusDraft {
		return 0, 0, ErrRaffleWrongStatus
	}

	csvReader := csv.NewReader(r)
	// Read header
	header, err := csvReader.Read()
	if err != nil {
		return 0, 0, fmt.Errorf("csv header: %w", err)
	}
	colIdx := csvColumnIndex(header)

	var imported, skipped int
	now := time.Now()

	for {
		record, err := csvReader.Read()
		if err == io.EOF {
			break
		}
		if err != nil {
			return imported, skipped, fmt.Errorf("csv row: %w", err)
		}

		twitchID := record[colIdx["twitch_user_id"]]
		displayName := record[colIdx["display_name"]]
		tier := int16(1)
		if idx, ok := colIdx["tier"]; ok && idx < len(record) {
			if v := record[idx]; v == "2" {
				tier = 2
			} else if v == "3" {
				tier = 3
			}
		}

		// Look up tachigo user by Twitch provider ID
		var provider models.AuthProvider
		if err := s.db.
			Where("provider = ? AND provider_id = ?", models.ProviderTwitch, twitchID).
			First(&provider).Error; err != nil {
			skipped++
			continue
		}

		entry := &models.RaffleEntry{
			RaffleID:     raffleID,
			UserID:       provider.UserID,
			TwitchUserID: twitchID,
			DisplayName:  displayName,
			Tier:         tier,
			IsEligible:   true,
			CreatedAt:    now,
		}
		if err := s.db.Create(entry).Error; err != nil {
			return imported, skipped, err
		}
		imported++
	}

	// Update status and snapshot time
	s.db.Model(raffle).Updates(map[string]interface{}{
		"status":      models.RaffleStatusSnapshotReady,
		"snapshot_at": now,
	})

	return imported, skipped, nil
}

func csvColumnIndex(header []string) map[string]int {
	m := make(map[string]int, len(header))
	for i, col := range header {
		m[strings.TrimSpace(strings.ToLower(col))] = i
	}
	return m
}
```

```go
// DrawNext randomly selects one eligible, undrawn entry and creates a RaffleDraw.
// On the first draw, raffle status transitions from snapshot_ready to live.
func (s *RaffleService) DrawNext(raffleID uuid.UUID) (*models.RaffleDraw, error) {
	raffle, err := s.GetByID(raffleID)
	if err != nil {
		return nil, err
	}
	if raffle.Status != models.RaffleStatusSnapshotReady && raffle.Status != models.RaffleStatusLive {
		return nil, ErrRaffleWrongStatus
	}

	// Pick a random eligible entry that hasn't been drawn
	var entry models.RaffleEntry
	if err := s.db.
		Where("raffle_id = ? AND is_eligible = ? AND drawn_at IS NULL", raffleID, true).
		Order("RANDOM()").
		First(&entry).Error; err != nil {
		if errors.Is(err, gorm.ErrRecordNotFound) {
			return nil, ErrNoEligibleEntries
		}
		return nil, err
	}

	// Count existing draws to determine draw_order
	var count int64
	s.db.Model(&models.RaffleDraw{}).Where("raffle_id = ?", raffleID).Count(&count)

	now := time.Now()
	draw := &models.RaffleDraw{
		RaffleID:       raffleID,
		EntryID:        entry.ID,
		DrawOrder:      int(count) + 1,
		Status:         models.DrawStatusPending,
		ClaimToken:     uuid.New(),
		ClaimExpiresAt: now.Add(time.Duration(raffle.ClaimDeadlineHours) * time.Hour),
		CreatedAt:      now,
	}

	tx := s.db.Begin()

	if err := tx.Create(draw).Error; err != nil {
		tx.Rollback()
		return nil, err
	}

	// Mark entry as drawn
	if err := tx.Model(&entry).Update("drawn_at", now).Error; err != nil {
		tx.Rollback()
		return nil, err
	}

	// Transition status to live on first draw
	if raffle.Status == models.RaffleStatusSnapshotReady {
		if err := tx.Model(raffle).Update("status", models.RaffleStatusLive).Error; err != nil {
			tx.Rollback()
			return nil, err
		}
	}

	tx.Commit()
	return draw, nil
}
```

```go
func (s *RaffleService) GetByClaimToken(token uuid.UUID) (*models.RaffleDraw, error) {
	var draw models.RaffleDraw
	if err := s.db.Where("claim_token = ?", token).First(&draw).Error; err != nil {
		if errors.Is(err, gorm.ErrRecordNotFound) {
			return nil, ErrClaimTokenInvalid
		}
		return nil, err
	}
	return &draw, nil
}

func (s *RaffleService) ClaimByToken(token uuid.UUID, recipientName, phone, addressJSON string) (*models.RaffleClaim, error) {
	draw, err := s.GetByClaimToken(token)
	if err != nil {
		return nil, err
	}
	if draw.Status == models.DrawStatusClaimed {
		return nil, ErrDrawAlreadyClaimed
	}
	if draw.Status == models.DrawStatusExpired {
		return nil, ErrDrawExpired
	}
	if time.Now().After(draw.ClaimExpiresAt) {
		s.db.Model(draw).Update("status", models.DrawStatusExpired)
		return nil, ErrDrawExpired
	}

	now := time.Now()
	claim := &models.RaffleClaim{
		DrawID:        draw.ID,
		RecipientName: recipientName,
		Phone:         phone,
		AddressJSON:   addressJSON,
		SubmittedAt:   now,
	}

	tx := s.db.Begin()
	if err := tx.Create(claim).Error; err != nil {
		tx.Rollback()
		return nil, err
	}
	if err := tx.Model(draw).Update("status", models.DrawStatusClaimed).Error; err != nil {
		tx.Rollback()
		return nil, err
	}
	tx.Commit()

	return claim, nil
}

func (s *RaffleService) Complete(raffleID uuid.UUID) error {
	raffle, err := s.GetByID(raffleID)
	if err != nil {
		return err
	}
	if raffle.Status != models.RaffleStatusLive {
		return ErrRaffleWrongStatus
	}
	return s.db.Model(raffle).Update("status", models.RaffleStatusCompleted).Error
}

func (s *RaffleService) ListDraws(raffleID uuid.UUID) ([]models.RaffleDraw, error) {
	var draws []models.RaffleDraw
	if err := s.db.
		Where("raffle_id = ?", raffleID).
		Order("draw_order ASC").
		Find(&draws).Error; err != nil {
		return nil, err
	}
	return draws, nil
}
```
